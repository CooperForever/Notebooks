# reset

git reset HEAD^        #回退所有内容到上一个版本
git reset HEAD~1       #回退内容到上一个版本
git reset HEAD^2       #回退内容到上上一个版本


git reset HEAD^ 文件名  #回退某个文件的版本到上一个版本
git reset 版本

# revert

用一次新提交取消前面的提交

git revert HEAD #撤销前一次提交
git revert HEAD^ #撤销前前一次提交
